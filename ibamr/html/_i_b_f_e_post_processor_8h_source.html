83  static inline void
84  EE_fcn(libMesh::TensorValue<double>& EE,
85  const libMesh::TensorValue<double>& FF,
86  const libMesh::Point& /*X*/,

104  static inline void cauchy_stress_from_PK1_stress_fcn(
105  libMesh::TensorValue<double>& sigma,
106  const libMesh::TensorValue<double>& FF,
107  const libMesh::Point& X,
108  const libMesh::Point& s,
109  libMesh::Elem* elem,
110  const std::vector<libMesh::NumericVector<double>*>& system_data,
111  double data_time,
112  void* ctx)
113  {
114  TBOX_ASSERT(ctx);
115  std::pair<IBTK::TensorMeshFcnPtr, void*>* PK1_stress_fcn_data =
116  static_cast<std::pair<IBTK::TensorMeshFcnPtr, void*>*>(ctx);
117  TBOX_ASSERT(PK1_stress_fcn_data);
118  IBTK::TensorMeshFcnPtr PK1_stress_fcn = PK1_stress_fcn_data->first;
119  void* PK1_stress_fcn_ctx = PK1_stress_fcn_data->second;
120  libMesh::TensorValue<double> PP;
121  PK1_stress_fcn(PP, FF, X, s, elem, system_data, data_time, PK1_stress_fcn_ctx);
122  sigma = PP * FF.transpose() / FF.det();
123  return;
124  } // cauchy_stress_from_PK1_stress_fcn

133  static inline void
134  deformed_material_axis_fcn(libMesh::VectorValue<double>& f,
135  const libMesh::TensorValue<double>& FF,
136  const libMesh::Point& /*X*/,
137  const libMesh::Point& /*s*/,
