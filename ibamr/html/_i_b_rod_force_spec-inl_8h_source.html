153 inline void IBRodForceSpec::packStream(SAMRAI::tbox::AbstractStream& stream)
154 {
155  const unsigned int num_rods = d_next_idxs.size();
156 #if !defined(NDEBUG)
157  TBOX_ASSERT(num_rods == d_material_params.size());
158 #endif
159  stream << static_cast<int>(num_rods);
160  stream.pack(&d_master_idx, 1);
161  stream.pack(&d_next_idxs[0], num_rods);
162  for (unsigned int n = 0; n < num_rods; ++n)
163  {
164  stream.pack(d_material_params[n].data(), NUM_MATERIAL_PARAMS);
165  }
166  return;
167 } // packStream
