163 inline void IBBeamForceSpec::packStream(SAMRAI::tbox::AbstractStream& stream)
164 {
165  const unsigned int num_beams = d_neighbor_idxs.size();
166 #if !defined(NDEBUG)
167  TBOX_ASSERT(num_beams == d_bend_rigidities.size());
168  TBOX_ASSERT(num_beams == d_mesh_dependent_curvatures.size());
169 #endif
170  std::vector<int> tmp_neighbor_idxs(2 * num_beams);
171  for (unsigned int k = 0; k < num_beams; ++k)
172  {
173  tmp_neighbor_idxs[2 * k] = d_neighbor_idxs[k].first;
174  tmp_neighbor_idxs[2 * k + 1] = d_neighbor_idxs[k].second;
175  }
176  stream << static_cast<int>(num_beams);
177  stream.pack(&d_master_idx, 1);
178  stream.pack(&tmp_neighbor_idxs[0], 2 * num_beams);
179  stream.pack(&d_bend_rigidities[0], 1 * num_beams);
180  for (unsigned k = 0; k < num_beams; ++k)
181  {
182  stream.pack(d_mesh_dependent_curvatures[k].data(), NDIM);
183  }
184  return;
185 } // packStream
