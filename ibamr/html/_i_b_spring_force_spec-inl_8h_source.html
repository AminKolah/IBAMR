154 inline size_t IBSpringForceSpec::getDataStreamSize() const
155 {
156  const unsigned int num_springs = d_slave_idxs.size();
157 #if !defined(NDEBUG)
158  TBOX_ASSERT(num_springs == d_force_fcn_idxs.size());
159  TBOX_ASSERT(num_springs == d_parameters.size());
160 #endif
161  size_t size = (2 + 2 * num_springs) * SAMRAI::tbox::AbstractStream::sizeofInt();
162  for (unsigned int k = 0; k < num_springs; ++k)
163  {
164  size += SAMRAI::tbox::AbstractStream::sizeofInt() +
165  d_parameters[k].size() * SAMRAI::tbox::AbstractStream::sizeofDouble();
166  }
167  return size;
168 } // getDataStreamSize

170 inline void IBSpringForceSpec::packStream(SAMRAI::tbox::AbstractStream& stream)
171 {
172  const unsigned int num_springs = d_slave_idxs.size();
173 #if !defined(NDEBUG)
174  TBOX_ASSERT(num_springs == d_force_fcn_idxs.size());
175  TBOX_ASSERT(num_springs == d_parameters.size());
176 #endif
177  stream << static_cast<int>(num_springs);
178  stream.pack(&d_master_idx, 1);
179  stream.pack(&d_slave_idxs[0], num_springs);
180  stream.pack(&d_force_fcn_idxs[0], num_springs);
181  for (unsigned int k = 0; k < num_springs; ++k)
182  {
183  const int num_parameters = d_parameters[k].size();
184  stream << num_parameters;
185  stream.pack(&d_parameters[k][0], num_parameters);
186  }
187  return;
188 } // packStream
