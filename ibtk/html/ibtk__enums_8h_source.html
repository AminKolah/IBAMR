79 inline MGCycleType string_to_enum<MGCycleType>(const std::string& val)
80 {
81  if (strcasecmp(val.c_str(), "V") == 0) return V_CYCLE;
82  if (strcasecmp(val.c_str(), "V_CYCLE") == 0) return V_CYCLE;

87  if (strcasecmp(val.c_str(), "F") == 0) return F_CYCLE;
