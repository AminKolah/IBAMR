48 inline SAMRAI::hier::Index<NDIM>
49 IndexUtilities::coarsen(const SAMRAI::hier::Index<NDIM>& i_fine,
50  const SAMRAI::hier::Index<NDIM>& ratio)
51 {
52  SAMRAI::hier::Index<NDIM> i_coarse;
53  for (unsigned int d = 0; d < NDIM; ++d)
54  {
55  i_coarse(d) = i_fine(d) < 0 ? (i_fine(d) + 1) / ratio(d) - 1 : i_fine(d) / ratio(d);
56  }
57  return i_coarse;
58 } // coarsen
