147 template <typename T>
148 inline T* LNode::getNodeDataItem() const
149 {
150  if (LIKELY(T::STREAMABLE_CLASS_ID < MAX_SIZE))
151  {
152  return static_cast<T*>(d_node_data_type_arr[T::STREAMABLE_CLASS_ID]);
153  }
154  else
155  {
156  const unsigned int node_data_sz = d_node_data.size();
157  T* ret_val = NULL;
158  Streamable* it_val;
159  unsigned int k;
160  for (k = 0; k < node_data_sz && !ret_val; ++k)
161  {
162  it_val = d_node_data[k];
163  if (it_val->getStreamableClassID() == T::STREAMABLE_CLASS_ID)
164  {
165  ret_val = static_cast<T*>(it_val);
166  }
167  }
168  return ret_val;
169  }
170 } // getNodeDataItem

235 inline void LNode::setupNodeDataTypeArray()
236 {
237  std::fill(
238  d_node_data_type_arr, d_node_data_type_arr + MAX_SIZE, static_cast<Streamable*>(NULL));
239  Streamable* it_val;
240  int class_id;
241  for (std::vector<SAMRAI::tbox::Pointer<Streamable> >::const_iterator cit =
242  d_node_data.begin();
243  cit != d_node_data.end();
244  ++cit)
245  {
246  it_val = *cit;
247  class_id = it_val->getStreamableClassID();
248  if (LIKELY(class_id < MAX_SIZE && !d_node_data_type_arr[class_id]))
249  {
250  d_node_data_type_arr[class_id] = it_val;
251  }
252  }
253  return;
254 } // setupNodeDataTypeArray
