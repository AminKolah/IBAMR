220  inline bool operator()(const LNodeIndex& lhs, const LNodeIndex& rhs)
221  {
222 #if !defined(NDEBUG)
223 #if ((NDIM > 3) || (NDIM < 1))
224  TBOX_ERROR("operator<(const LNodeIndex&,const LNodeIndex&): not implemented for NDIM=="
225  << NDIM << endl);
226 #endif
227 #endif
228  const double* const X_lhs =
229  &(*d_X_ghosted_local_form_array)[lhs.getLocalPETScIndex()][0];
230  const double* const X_rhs =
231  &(*d_X_ghosted_local_form_array)[rhs.getLocalPETScIndex()][0];
232  return ((X_lhs[0] < X_rhs[0])) ||
233 #if (NDIM > 1)
234  ((X_lhs[0] == X_rhs[0]) && (X_lhs[1] < X_rhs[1])) ||
235 #if (NDIM > 2)
236  ((X_lhs[0] == X_rhs[0]) && (X_lhs[1] == X_rhs[1]) && (X_lhs[2] < X_rhs[2])) ||
237 #endif
238 #endif
239  ((X_lhs[0] == X_rhs[0]) &&
240 #if (NDIM > 1)
241  (X_lhs[1] == X_rhs[1]) &&
242 #if (NDIM > 2)
243  (X_lhs[2] == X_rhs[2]) &&
244 #endif
245 #endif
246  (lhs.getLagrangianIndex() < rhs.getLagrangianIndex()));
247  } // operator()
248 
249  inline bool operator()(const LNodeIndex* lhs, const LNodeIndex* rhs)
250  {
251  return (*this)(*lhs, *rhs);
252  } // operator()
253 
254 private:
