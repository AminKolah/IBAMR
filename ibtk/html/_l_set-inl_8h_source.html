227 template <class T>
228 inline void LSet<T>::unpackStream(SAMRAI::tbox::AbstractStream& stream,
229  const SAMRAI::hier::IntVector<NDIM>& offset)
230 {
231  d_offset = offset;
232  int num_idx;
233  stream.unpack(&num_idx, 1);
234  d_set.resize(num_idx);
235  for (unsigned int k = 0; k < d_set.size(); ++k)
236  {
237  d_set[k] = new T(stream, offset);
238  }
239  typename LSet<T>::DataSet(d_set).swap(d_set); // trim-to-fit
240  return;
241 } // unpackStream
