166  static inline void ib_4_interp_fcn(const double r, double* const w)
167  {
168  const double q = sqrt(1.0 + 4.0 * r * (1.0 - r));
169  w[0] = 0.125 * (3.0 - 2.0 * r - q);
170  w[1] = 0.125 * (3.0 - 2.0 * r + q);
171  w[2] = 0.125 * (1.0 + 2.0 * r + q);
172  w[3] = 0.125 * (1.0 + 2.0 * r - q);
173  return;
174  } // ib_4_interp_fcn
